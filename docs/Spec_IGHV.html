Multivariate linear regresson with lasso penalty and 3-fold cross validation using misclassification measurement

repetition <- 100
y <- patDF$IGHV
resMat <- lapply(seq_len(repetition), function(n){
    smpIdx <- c(sample(order(match(y, "M"))[1:sum(y == "M")], sum(y == "U"), replace = TRUE), order(match(y, "M"))[(sum(y == "M") + 1):length(y)])
    X <- as.matrix(assay.fil[smpIdx, ])
    cv.lasso <- cv.glmnet(x = X,
                          y = y[smpIdx], 
                          family = "binomial",
                          alpha = 1,
                          nfolds = 3,
                          type.measure = "class")
    coef(cv.lasso, s = cv.lasso$lambda.min) %>%
      as.matrix() %>%
      rbind("errors" = cv.lasso$cvm[cv.lasso$index["min", ]])
  }) %>% do.call(cbind, .)
colnames(resMat) <- seq_len(repetition)

errTab <- as_tibble(resMat["errors", ], rownames = "repetition")
coefMat <- resMat[-c(1, nrow(resMat)), ]

NR <- 30
cutoff <- rowSums(coefMat != 0) %>% 
  as_tibble(rownames = "geneID") %>% 
  filter(value != 0) %>%
  group_by(value) %>%
  summarise(count = length(value)) %>%
  ungroup() %>%
  arrange(desc(value)) %>% 
  mutate(cum = cumsum(count), dif = cum - NR ) %>%
  filter(abs(dif) == min(abs(dif))) %>%
  pull(value)
# frequency vs mean plot
sumDF <- apply(coefMat, 1, function(v) c(mean = mean(v[v != 0]), sd = sd(v[v != 0]))) %>% 
  ifelse(is.nan(.) | is.na(.), 0, .) %>%
  t() %>% 
  as_tibble(rownames = "geneID") %>%
  left_join(as_tibble(rownames = "geneID", rowSums(coefMat != 0) / repetition * 100))  %>%
  setNames(c("geneID", "mean", "sd", "frequency")) %>%
  mutate(coefficient = case_when(mean > 0 & frequency >= cutoff ~ "positive", 
                                 mean == 0 | frequency < cutoff ~ "not selected",
                                 mean < 0 & frequency >= cutoff ~ "negative"),
         delabel = ifelse(coefficient == "not selected", NA, geneID))  %>%
  arrange(mean)
Joining with `by = join_by(geneID)`
sumDF %>% ggplot(aes(x = mean, y = frequency, col = coefficient)) +
  geom_point() +
  scale_color_manual(values = c("dodgerblue3", "gray50", "firebrick3")) +
  geom_hline(yintercept = cutoff, lty = 4, col = "black", lwd = 0.8) +
  geom_vline(xintercept = 0, lty = 4, col = "black", lwd = 0.8) +
  geom_label(aes(label = delabel), size = 3) +
  labs(x = "Average coefficients", y = "Frequency (%)") +
  theme_bw() + 
  theme(axis.text = element_text(size = 16),
        axis.title = element_text(size = 20, face = "bold"),
        legend.title = element_text(size = 16),
        legend.text = element_text(size = 14))
