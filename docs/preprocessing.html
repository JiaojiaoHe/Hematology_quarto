smpDF$diag <- stringi::stri_replace_all_regex(smpDF$diag, 
                       pattern = c(" \\+ ", ", ", " \\(", "\\) ", "\\)", "\\?", "/"),
                       replacement = c(rep("_", 3), rep("", 3), "_or_"),
                       vectorize = FALSE) %>%
  str_replace("\\+", "_") %>%
  str_replace_all(" ", "_")

DIANN SE

assay.diann <- pivot_wider(diannL, names_from = smpID, values_from = count) %>%
  filter(!(protID %in% ctrlProts | str_detect(protID, "Y-FGCZCont"))) %>% 
  arrange(protID) %>%
  column_to_rownames("protID") %>% 
  select(rownames(smpDF)) %>%
  as.matrix()
protAnno.diann <- left_join(as_tibble(assay.diann, rownames = "protID"), protAnno) %>% 
  select(colnames(protAnno)) %>%
  make_unique("geneID", "protID", delim = ";")
Joining with `by = join_by(protID)`
rownames(assay.diann) <- protAnno.diann$name
rownames(protAnno.diann) <- protAnno.diann$name
seDiann <- SummarizedExperiment(assays = assay.diann, 
                                colData = smpDF, 
                                rowData = protAnno.diann)
assayNames(seDiann) <- "count"
assays(seDiann)$log2 <- log2(assay.diann)
save(seDiann, file = "output/seDiann.RData")
