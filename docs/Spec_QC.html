missing rate

missTab <- assay(seSpec) %>% as.data.frame() %>%
  mutate(sampleNr = rowSums(!is.na(.)), 
         missRate = rowSums(is.na(.))/ncol(.))

missPropTab <- missTab  %>%
  group_by(missRate, sampleNr) %>%
  summarise(protNr = length(missRate), 
            proportion = protNr/nrow(.)) %>% ungroup()

plot_frequency(seSpec) +

upset plot

patTab <- colData(seSpec) %>% as_tibble(rownames = "smpID") %>%
    mutate(diagSimple = ifelse(diag %in% c("CLL", "healthy", "MCL", "HCL", "SMZL", "MZL"), diag, "other")) 
presTab <- assay(seSpec) %>% as_tibble(rownames = "protID") %>%
  pivot_longer(-protID, names_to = "smpID", values_to = "count") %>%
  left_join(patTab, by = c("smpID" = "smpID")) %>%
  group_by(diagSimple, protID) %>%
  summarise(presence = ifelse(sum(is.na(count)) == 0 , 1, 0)) %>%
  ungroup() %>%
  pivot_wider(names_from = diagSimple, values_from = presence) %>%
  column_to_rownames("protID")
`summarise()` has grouped output by 'diagSimple'. You can override using the
`.groups` argument.
setVars <- c("CLL", "healthy", "MCL", "HCL", "SMZL", "MZL", "other")
upset(presTab,
      sets = setVars,
      order.by = "freq",
        main.bar.color = "violetred4",
        sets.bar.color = "turquoise4",
        matrix.color = "slateblue4",
        shade.color = "wheat4")

allImp <- lapply(c("qrilc", "minProb", "knn"), function(imp){
  assays(seSub)[[imp]] %>%
    tibble(Mean = rowMeans(., ), 
           `Proportion of Missing` = rowSums(is.na(assay(seSub)))/ncol(.)) %>%
    mutate(Imputation = imp)
}) %>% do.call(rbind, .)

plot1 <- assays(seSub)$vsn %>% t() %>%
  plot_meanMiss(bins = 100, color = "cyan") + scale_fill_viridis_c(option = "C")
plot1 + geom_smooth(allImp, mapping = aes(y = `Proportion of Missing`, x = Mean, col = Imputation), method = 'loess', se = FALSE) + facet_wrap(~Imputation, ncol = 2) + theme_bw()
