adjusted p values

# correlation test and anova
pvalTab <- lapply(batVec, function(fac){
  print(fac)
  sapply(colnames(pcRes$x), function(pc){
    ifelse(fac == "pelletSize",
           cor.test(pcTab[[pc]], pcTab$pelletSize, method = "kendall", exact = FALSE)$p.value,
           kruskal.test(as.formula(paste0(pc, "~", fac)), pcTab)$p.value)
    }) %>% p.adjust("BH")
}) %>% do.call(rbind, .)

[1] "sampleType"
[1] "patID"

pvalTab <- lapply(batVec, function(fac){
  print(fac)
  sapply(colnames(pcRes$x), function(pc){
    ifelse(fac == "pelletSize",
           cor.test(pcTab[[pc]], pcTab$pelletSize, method = "kendall")$p.value,

  separate(col = "PC", into = c("a", "PC"), sep = "PC") %>%
