upset plot

patTab <- colData(seSpec) %>% as_tibble(rownames = "smpID") %>%
    mutate(diagSimple = ifelse(diag %in% c("CLL", "healthy", "MCL", "HCL", "SMZL", "MZL"), diag, "other")) 
presTab <- assay(seSpec) %>% as_tibble(rownames = "protID") %>%
  pivot_longer(-protID, names_to = "smpID", values_to = "count") %>%
  left_join(patTab, by = c("smpID" = "smpID")) %>%
  group_by(diagSimple, protID) %>%
  summarise(presence = ifelse(sum(is.na(count)) == 0 , 1, 0)) %>%
  ungroup() %>%
  pivot_wider(names_from = diagSimple, values_from = presence) %>%
  column_to_rownames("protID")

# anova
batList <- list("pelSimple" = pelNum, "sampleType" = stpNum, "patID" = patNum, "set" = setNum)
pvalTab <- lapply(names(batList), function(fac){
  print(fac)
  data <- pcTab[pcTab[[fac]] %in% names(batList[[fac]][batList[[fac]] > 1]), ]
  sapply(colnames(pcRes$x), function(pc){
    res_aov <- aov(as.formula(paste0(pc, "~", fac)), data = data) 
    summary(res_aov)[[1]][["Pr(>F)"]][1]
    })
}) %>% do.call(rbind, .)
