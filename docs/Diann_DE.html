Generalized Linear Model

remove “S454563” that perhaps is a mis-annotation sample

outId <- which(rownames(cll2hcl.mat) %in% "S454563")
cll2hcl.mat <- cll2hcl.mat[-outId, ]
y <- ifelse(colData(deList$cll2hcl)[-outId, ]$diag == "CLL", 1, 0)

# run lm 
set.seed(1374)
epochs <- 100
system.time({resMat <- lapply(seq_len(epochs), function(n){
    smpIdx <- c(sample(order(match(y, 1))[1:sum(y)], sum(!y), replace = TRUE), order(match(y, 1))[(sum(y)+1):length(y)])
    X <- cll2hcl.mat[smpIdx, ]
    cv.lasso <- cv.glmnet(x = X,
                          y = y[smpIdx], 
                          family = "binomial",
                          alpha = 1,
                          nfolds = floor(sum(!y)/3),
                          type.measure = "deviance")
    coef(cv.lasso, s = cv.lasso$lambda.min) %>% 
      as.matrix() %>%
      rbind("errors" = cv.lasso$cvm[cv.lasso$index["min", ]])
  }) %>% do.call(cbind, .)
})
