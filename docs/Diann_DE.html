Generalized Linear Model

remove “S454563” that perhaps is a mis-annotation sample

# run lm 
set.seed(1374)
repetition <- 100
system.time({resMat <- lapply(seq_len(repetition), function(n){
    smpIdx <- c(sample(order(match(y, 1))[1:sum(y)], sum(!y), replace = TRUE), order(match(y, 1))[(sum(y)+1):length(y)])
    X <- cll2hcl.mat[smpIdx, ]
    cv.lasso <- cv.glmnet(x = X,
                          y = y[smpIdx], 
                          family = "binomial",
                          alpha = 1,
                          nfolds = sum(!y),
                          type.measure = "class")
    coef(cv.lasso, s = cv.lasso$lambda.min) %>% 
      as.matrix() %>%
      rbind("errors" = cv.lasso$cvm[cv.lasso$index["min", ]])
  }) %>% do.call(cbind, .)
})
