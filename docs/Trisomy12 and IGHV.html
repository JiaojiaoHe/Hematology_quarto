Trisomy12

NR <- 30
repetition <- 100
y <- patDF$trisomy12
resMat <- lapply(seq_len(repetition), function(n){
    smpIdx <- c(sample(order(match(y, 0))[1:sum(y == 0)], sum(y == 1), replace = TRUE), order(match(y, 0))[(sum(y == 0)+1):length(y)])
    X <- assay.fil[smpIdx, ]
    cv.lasso <- cv.glmnet(x = X,
                          y = y[smpIdx], 
                          family = "binomial",
                          alpha = 1,
                          nfolds = 3,
                          type.measure = "auc")
    coef(cv.lasso, s = cv.lasso$lambda.min) %>% 
      as.matrix() %>%
      rbind("errors" = cv.lasso$cvm[cv.lasso$index["min", ]])
  }) %>% do.call(cbind, .)
colnames(resMat) <- seq_len(repetition)
coefMat <- resMat[-c(1, nrow(resMat)), ]

cutoff <- rowSums(coefMat != 0) %>% 
  as_tibble(rownames = "protID") %>% 
  filter(value != 0) %>%
  group_by(value) %>%
  summarise(count = length(value)) %>%
  ungroup() %>%
  arrange(desc(value)) %>% 
  mutate(cum = cumsum(count), dif = cum - NR ) %>%
  filter(abs(dif) == min(abs(dif))) %>%
  pull(value)
# frequency vs mean plot
sumDF <- apply(coefMat, 1, function(v) c(mean = mean(v[v != 0]), sd = sd(v[v != 0]))) %>% 
  ifelse(is.nan(.) | is.na(.), 0, .) %>%
  t() %>% 
  as_tibble(rownames = "protID") %>%
  left_join(as_tibble(rownames = "protID", rowSums(coefMat != 0) / repetition * 100))  %>%
  setNames(c("protID", "mean", "sd", "frequency")) %>%
  left_join(geneDF) %>%
  mutate(coefficient = case_when(mean > 0 & frequency >= cutoff ~ "positive", 
                                 mean == 0 | frequency < cutoff ~ "not selected",
                                 mean < 0 & frequency >= cutoff ~ "negative"),
         delabel = ifelse(coefficient == "not selected", NA, hgnc_symbol))  %>%
  arrange(mean)
Joining with `by = join_by(protID)`

sumDF %>% ggplot(aes(x = mean, y = frequency, col = coefficient)) +
  geom_point() +
  scale_color_manual(values = c("dodgerblue3", "gray50", "firebrick3")) +
  geom_hline(yintercept = cutoff, lty = 4, col = "black", lwd = 0.8) +
  geom_vline(xintercept = 0, lty = 4, col = "black", lwd = 0.8) +
  labs(y = "Frequency (%)") +
  geom_label(aes(label = delabel), size = 3) +
